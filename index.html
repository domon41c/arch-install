1. Підключення до інтернету.

iwctl: station wlan0 connect "вай-фай"

2. Розділи диску

lsblk
cfdisk /dev/диск

3. Монтіруєм диски та стираємо їх.

lsblk
mkfs.fat -F32 /dev/бут-розділ
mkfs.ext4 /dev/корінь-розділ

4. Готовимо до установки.

mount /dev/корінь-розділ /mnt
mkdir /mnt/boot
mount /dev/бут-розділ /mnt/boot

5. Установка.

pacstrap /mnt base linux linux-firmware
genfstab -U /mnt >> /mnt/etc/fstab

6. Вход і юзери

arch-chroot /mnt
pacman -S nano sudo dhcpcd wpa_supplicant wireless_tools networkmanager
systemctl enable NetworkManager.service
passwd
useradd -m -g users -G wheel -s /bin/bash domon
passwd domon

7. Права юзера

EDITOR-nano visudo
строчка: domon ALL=(ALL) ALL

8. Grub установка

pacman -S grub efibootmgr
mkdir /boot/efi
mount /dev/бут-розділ /boot/efi
grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=GRUB --removable
grub-mkconfig -o /boot/grub/grub.cfg

9. Кінець
exit
umount -R /mnt
reboot
